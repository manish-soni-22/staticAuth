Frontend: Send Login Request to Backend // Example (React/Vue/Plain JS) const login = async () => { const response = await fetch("https://your-api.com/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ username: "user@example.com", password: "YourPassword123" }) }); const data = await response.json(); console.log("Tokens:", data); }; 2. Backend: Authenticate with Cognito using AWS SDK const AWS = require("aws-sdk"); AWS.config.update({ region: "us-east-1" }); // Set your region const cognito = new AWS.CognitoIdentityServiceProvider(); async function authenticateUser(username, password) { const params = { AuthFlow: "USER_PASSWORD_AUTH", // or ADMIN_NO_SRP_AUTH if you use admin creds ClientId: "your_cognito_app_client_id", // Must have "allow user password auth" AuthParameters: { USERNAME: username, PASSWORD: password } }; try { const result = await cognito.initiateAuth(params).promise(); return result.AuthenticationResult; // Contains AccessToken, IdToken, RefreshToken } catch (err) { throw new Error("Authentication failed: " + err.message); } } 3. Return Tokens to Client app.post("/login", async (req, res) => { const { username, password } = req.body; try { const tokens = await authenticateUser(username, password); res.json({ accessToken: tokens.AccessToken, idToken: tokens.IdToken, refreshToken: tokens.RefreshToken }); } catch (err) { res.status(401).json({ error: err.message }); } }); üõ°Ô∏è Best Practices Always use HTTPS between frontend and backend.
